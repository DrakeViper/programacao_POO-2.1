class Veiculo {
  String marca;
  String modelo;
  int ano;

  Veiculo(this.marca, this.modelo, this.ano);

  void exibirFicha() {
    print('Marca: $marca');
    print('Modelo: $modelo');
    print('Ano: $ano');
  }
}

class Carro extends Veiculo {
  int quantidadePortas; // só do carro

  Carro(String marca, String modelo, int ano, this.quantidadePortas)
      : super(marca, modelo, ano); // manda os 3 pro pai

  @override
  void exibirFicha() {
    super.exibirFicha(); // chama o do pai primeiro
    print('Portas: $quantidadePortas');
  }
}

void main() {
  Carro c = Carro('Toyota', 'Corolla', 2023, 4);
  c.exibirFicha();
}
